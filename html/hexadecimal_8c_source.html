20void hexadecimal(t_sm *machine)
21{
22 char str[42];
23 unsigned long long num;
24 int base;
25
26 ft_bzero(str, 42);

28 num = unsign_va_arg(machine);
29 if (!((machine->flag & BIT_PREC) && (machine->prec == 0) && (num == 0)))
30 {
31 if ((machine->flag & BIT_HASH) != FALSE)
32 {
33 adjust_hash(str, machine);
34 itoa_buff(num, &str[2], base, machine);
35 }
36 else
37 itoa_buff(num, str, base, machine);
38 }
39 formalize(str, machine);
40}
