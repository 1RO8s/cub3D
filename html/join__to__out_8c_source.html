14#include "libft.h"

16char *join_to_out(const char *s1, const char *s2, size_t len2)
17{
18 char *str;
19 size_t len1;
20
21 if ((s1 == NULL) || (s2 == NULL))
22 return (NULL);
23 len1 = ft_strlen(s1);
24 str = ft_strnew(len1 + len2);
25 if (str == NULL)
26 return (NULL);
27 ft_strlcpy(str, s1, len1 + 1);
28 ft_memcpy(&str[len1], s2, len2);
29 free((void *)s1);
30 return (str);
31}
