13# Build Libraries : nest of Cub3D
14# target -> MiniLibX, Libft, ft_printf
15
16# Makefile Option : Disable '--print-directory'
17MAKEFLAGS += --no-print-directory
18
19# Directories
20LIBFT_DIR = libft
21LIBFTPRINTF_DIR = ft_printf
22LIBMLX_LINUX_DIR = minilibx-linux
23LIBMLX_MAC_DIR = minilibx_mms_20200219
24LIBMLXOPENGL_DIR = libmlxopengl
25RESOURCE_DIR = ../resource
26LIBDEBUG_DIR = debug
27LIB_DIRS = $(LIBFT_DIR) $(LIBFTPRINTF_DIR)
28
29# Resource files
30RESOURCE_MLX_LINUX = minilibx-linux.tgz
31RESOURCE_MLX_MAC = minilibx_mms_20200219_beta.tgz
32RESOURCE_MLX_OPENGL = minilibx_opengl.tgz
33
34# Library name
35LIBFT = $(LIBFT_DIR)/libft.a
36LIBFTPRINTF = $(LIBFTPRINTF_DIR)/libftprintf.a
37LIBDEBUG = $(LIBDEBUG_DIR)/libdebug.a
38
39# Get OS type for choosing minilibx library
40OS := $(shell uname)
41# To Build MLX on macOS
42ifeq ($(OS), Darwin)
43RESOURCE_MLX := $(RESOURCE_MLX_LINUX)
44LIBMLX_DIR = $(LIBMLX_LINUX_DIR)
45LIBMLX = $(LIBMLX_DIR)/libmlx_Darwin.a
46endif
47# To Build MLX on Linux
48ifeq ($(OS), Linux)
49RESOURCE_MLX := $(RESOURCE_MLX_LINUX)
50LIBMLX_DIR = $(LIBMLX_LINUX_DIR)
51LIBMLX = $(LIBMLX_DIR)/libmlx_Linux.a
52endif
53
54LIBS = $(LIBMLX) $(LIBFT) $(LIBFTPRINTF)
55
56# Build target (Review)
57NAME = start $(LIBMLX)
58
59# Include configuration file
60include ../config/config.mk
61
62# Enable Develop mode
63DEV_ON ?= 0 # Default value is 0 if not specified
64
65ifeq ($(DEV_ON), 1)
66NAME = start submodules $(LIBMLX)
67endif
68
69# Default target
70all: $(NAME)
71 @for name in $(LIB_DIRS); do \
72 make -C $$name; \
73 done
74.PHONY: all
75
76# Enabel Debug mode
77ifeq ($(DEBUG_ON), 1)
78LIB_DIRS += $(LIBDEBUG_DIR)
79NAME += debug
80LIBS += $(LIBDEBUG)
81endif
82
83debug:
84ifeq ($(shell test -d $(LIBDEBUG_DIR) && echo exist),)
85 @echo "Please clone the debug repository"
86 @exit 1
87endif
88 @echo "Building in debug mode..."
89 make -C lib/debug
90
91PROJECT_DIR = $(notdir $(CURDIR))
92start:
93 @echo "${YELLOW}Starting build process for '${PROJECT_DIR}'...${NC}"
94.PHONY: start
95
96# Build MiniLibX
97$(LIBMLX):
98 make extract_mlx
99 make install_dependencies
100 @echo "${YELLOW}Starting build process for '${LIBMLX_DIR}'...${NC}"
101 make -C $(LIBMLX_DIR)
102 @echo "${GREEN}Successfully created archive: $@${NC}"
103
104# Extract from resource (minilibx-linux.tgz minilibx_opengl.tgz)
105extract_mlx:
106 @./extract_mlx.sh $(RESOURCE_DIR)/$(RESOURCE_MLX)
107.PHONY: extract_mlx
108
109# Check dependencies installed
110install_dependencies:
111 @./install_dependency.sh
112.PHONY: install_dependencies
113
114# Update submodules
115SUBMODULE_DIR := $(LIBFT_DIR)
116submodules:
117 @command -v git >/dev/null 2>&1 \
118 || { echo >&2 "git is not installed. Aborting."; exit 1; }
119 @./initialize_submodules.sh
120.PHONY: submodules
121
122## Clean target
123clean:
124 @echo "${RED}Cleaning object files of Libraries...${NC}"
125 @for name in $(LIB_DIRS); do \
126 make -C $$name clean; \
127 done
128 @if [ -d "$(LIBMLX_DIR)" ]; then \
129 make -C $(LIBMLX_DIR) clean; \
130 else \
131 echo "Warning: $(LIBMLX_DIR) directory does not exist, skipping."; \
132 fi
133.PHONY: clean
134
135# Clean and remove library target
136fclean: clean
137 @echo "${RED}Removing archive file...${NC}"
138 rm -f $(LIBS)
139 @echo "${GREEN}All archive file removed.${NC}"
140.PHONY: fclean
141
142## Rebuild target
143re: fclean all
144.PHONY: re
145
146# Color Definitions
147RED=\033[0;31m
148GREEN=\033[0;32m
149YELLOW=\033[0;33m
150NC=\033[0m # No Color
