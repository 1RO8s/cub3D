114# Build target
115NAME = cub3D

117# vpath for serching source files in multiple directories
118vpath %.c $(SRCS_DIR)

120# Compiler
121CC = cc
122CFLAGS = -Wall -Wextra -Werror
123CF_OPTIMIZE = -O3
124CF_ASAN = -g -fsanitize=address
125CF_GENERATE_DEBUG_INFO = -g
126CF_INC = -I$(INC_DIR) -I$(LIBFT_DIR) -I$(LIBFTPRINTF_DIR)/includes \
127 -I$(LIBMLX_DIR) -I$(LIBDEBUG_DIR)/includes
128CF_DEP = -MMD -MP -MF $(@:$(OBJ_DIR)/%.o=$(DEP_DIR)/%.d)
129
130# Get OS type for choosing API
131OS := $(shell uname)
132# To link internal macOS API
133ifeq ($(OS), Darwin)
134LIBMLX := $(LIBMLX_DIR)/libmlx_Darwin.a
135CF_API = -L./lib/minilibx-linux -lmlx_Darwin -L/usr/X11R6/lib -lX11 -lXext -framework OpenGL -framework AppKit
136endif
137# To link internal Linux API
138ifeq ($(OS), Linux)
139LIBMLX := $(LIBMLX_DIR)/libmlx_Linux.a
140CF_API = -L$(LIBMLX_DIR) -lmlx_Linux -L/usr/lib -lXext -lX11 -lm -lz
141endif

143# Rules for building object files
144$(OBJ_DIR)/%.o: %.c
145 @mkdir -p $(OBJ_DIR)
146 @mkdir -p $(DEP_DIR)
147 $(CC) $(CFLAGS) $(CF_INC) $(CF_OPTIMIZE) $(CF_DEP) -c $< -o $@
148
149$(DEP_DIR)/%.d: %.c
150 @mkdir -p $(DEP_DIR)

152# Default target
153all: start build_lib $(NAME) end display_art
154.PHONY: all
155
156# Out starting message
157start:
158 @$(if $(filter 1, $(DEBUG_ON)), echo "Debug Mode ON : Enable build process for debug library.")
159 @$(if $(filter 1, $(DEV_ON)), echo "Develop mode ON : In the 'main' branch.")
160 @echo "${YELLOW}Starting build process for '${NAME}'...${NC}"
161.PHONY: start
162
163# Build libraries
164$(LIBS): build_lib
165
166build_lib:
167 @$(if $(filter 1, $(DEBUG_ON)), \
168 make -C $(LIB_DIR) DEBUG_ON=$(DEBUG_ON), \
169 make -C $(LIB_DIR))
170 @echo "${YELLOW}Build process completed for '${LIBS}'.${NC}"
171.PHONY: build_lib
172
173# Out ending message
174end:
175 @echo "${YELLOW}Build process completed $(NAME).${NC}"
176.PHONY: end
177
178display_art:
179 $(ASCII_ART)
180.PHONY: display_art
181
182# Target
183$(NAME): $(LIBS) $(DEPS) $(OBJS)
184 $(CC) $(CFLAGS) $(OBJS) $(LIBS) $(CF_API) -o $(NAME)
185 @echo "${GREEN}Successfully created execute: $@${NC}"

187# Option : make rules

189# Address sanitizer mode make rule
190asan: fclean
191 make WITH_ASAN=1
192.PHONY: asan

194# Leak check
195check: fclean
196 make WITH_VALGRIND=1
197 $(VALGRIND_USAGE)
198.PHONY: check
199
200# Clean target
201clean:
202 @echo "${RED}Cleaning object files of '${NAME}'...${NC}"
203 rm -rf $(OBJ_DIR) $(DEP_DIR)
204 @for name in $(LIB_DIRS); do \
205 if [ -d "$$name" ]; then \
206 make -C $$name clean; \
207 else \
208 echo "Warning: $$name directory does not exist, skipping."; \
209 fi; \
210 done
211.PHONY: clean
212# make -C $$name clean; \
213
214# Clean and remove library target
215fclean: clean
216 @echo "${RED}Removing archive file...${NC}"
217 rm -f $(LIBS)
218 rm -f $(NAME)
219 @echo "${GREEN}Archive file removed.${NC}"
220.PHONY: fclean
221
222# Rebuild target
223re: fclean all
224.PHONY: re
